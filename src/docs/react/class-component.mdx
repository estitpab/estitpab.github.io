### Structure du composant

```jsx title=classComponent.js
import React from 'react';

class Car extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      brand: "Peugeot",
      model: "308",
    };
  };

  render() {
    return (
      <div>
        <h1>My {this.state.brand}</h1>
        <p>
          It is a {this.state.model}.
        </p>
      </div>
    );
  };
};

export default Car;
```
